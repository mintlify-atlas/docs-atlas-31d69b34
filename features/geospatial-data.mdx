## Overview

SMyEG includes a powerful **geospatial visualization system** built with Leaflet, enabling interactive mapping of forest patrimony data with support for multiple base maps, layer management, and drawing tools.

## Leaflet Integration

The GIS system uses React Leaflet for map rendering with custom controllers:

### Loading Layers

Layers are fetched from the API with bounding box filtering:

```typescript src/components/GeoDashboardMapClient.tsx
const loadLayer = useCallback(async () => {
  setLoadingLayer(true);
  try {
    const response = await fetch(
      `/api/forest/geo/layers/nivel4?bbox=${WORLD_BBOX}`,
      { cache: "no-store" }
    );
    const payload = await response.json();

    if (payload.success && payload.data) {
      const layerData = payload.data;
      setLayers((previous) => {
        const existing = previous.find((layer) => 
          layer.id === "rodales-activos"
        );
        const baseLayer = createLayer(
          "Rodales activos (Nivel 4)", 
          layerData, 
          existing
        );
        baseLayer.id = "rodales-activos";
        const others = previous.filter((layer) => 
          layer.id !== "rodales-activos"
        );
        return applyPersistedSettings([baseLayer, ...others], settings);
      });
    }
  } catch (error) {
    // Error handling
  } finally {
    setLoadingLayer(false);
  }
}, []);
```

### Distance Calculation

Uses the Haversine formula for accurate distance calculations:

```typescript src/components/GeoDashboardMapClient.tsx
function haversineDistanceMeters(a: PointerCoords, b: PointerCoords) {
  const toRadians = (value: number) => (value * Math.PI) / 180;
  const earthRadiusMeters = 6371000;

  const lat1 = toRadians(a.lat);
  const lat2 = toRadians(b.lat);
  const deltaLat = toRadians(b.lat - a.lat);
  const deltaLng = toRadians(b.lng - a.lng);

  const h = Math.sin(deltaLat / 2) ** 2
    + Math.cos(lat1) * Math.cos(lat2) * Math.sin(deltaLng / 2) ** 2;

  return 2 * earthRadiusMeters * Math.asin(Math.sqrt(h));
}

function formatDistance(distanceMeters: number) {
  if (distanceMeters < 1000) return `${distanceMeters.toFixed(1)} m`;
  return `${(distanceMeters / 1000).toFixed(3)} km`;
}
```

### Area Calculation

Uses Turf.js for polygon area calculations:

```typescript src/components/GeoDashboardMapClient.tsx
import { area as turfArea, polygon as turfPolygon } from "@turf/turf";

const drawAreaM2 = useMemo(() => {
  if (drawPoints.length < 3) return 0;

  const ring = drawPoints.map((point) => [point.lng, point.lat]);
  const first = ring[0];
  const last = ring[ring.length - 1];
  
  if (first && last && (first[0] !== last[0] || first[1] !== last[1])) {
    ring.push([first[0], first[1]]); // Close the ring
  }

  try {
    const polygon = turfPolygon([ring]);
    return turfArea(polygon);
  } catch {
    return 0;
  }
}, [drawPoints]);

function formatArea(areaM2: number) {
  if (areaM2 >= 10000) return `${(areaM2 / 10000).toFixed(3)} ha`;
  return `${areaM2.toFixed(1)} m²`;
}
```

## Map Controllers

### Resize Controller

Ensures the map resizes correctly when the layout changes:

```typescript src/components/GeoDashboardMapClient.tsx
function MapResizeController({ resizeToken }: { resizeToken: string }) {
  const map = useMap();

  useEffect(() => {
    const run = () => map.invalidateSize({ animate: false });
    
    run();
    const raf = window.requestAnimationFrame(run);
    const timeout = window.setTimeout(run, 180);

    return () => {
      window.cancelAnimationFrame(raf);
      window.clearTimeout(timeout);
    };
  }, [map, resizeToken]);

  return null;
}
```

### Interaction Controller

Handles mouse events for drawing and coordinate capture:

```typescript src/components/GeoDashboardMapClient.tsx
function MapInteractionController({
  editMode,
  isCoordinateCaptureEnabled,
  onPointerMove,
  onMapClick,
  onMapMouseUp
}: Props) {
  useMapEvents({
    mousemove: (event) => {
      onPointerMove({ lat: event.latlng.lat, lng: event.latlng.lng });
    },
    mouseup: () => {
      onMapMouseUp();
    },
    click: (event) => {
      if (editMode === "none" && !isCoordinateCaptureEnabled) return;
      onMapClick({ lat: event.latlng.lat, lng: event.latlng.lng });
    }
  });

  return null;
}
```

### View Persistence

Automatically saves and restores the map view:

```typescript src/components/GeoDashboardMapClient.tsx
function MapViewPersistenceController({ onViewChange }: Props) {
  const map = useMap();

  useMapEvents({
    moveend: () => {
      const center = map.getCenter();
      onViewChange({ 
        lat: center.lat, 
        lng: center.lng, 
        zoom: map.getZoom() 
      });
    },
    zoomend: () => {
      const center = map.getCenter();
      onViewChange({ 
        lat: center.lat, 
        lng: center.lng, 
        zoom: map.getZoom() 
      });
    }
  });

  return null;
}
```

## Search and Filtering

### Hierarchical Filtering

Filter features by forest patrimony hierarchy:

```typescript src/components/GeoDashboardMapClient.tsx
const filteredVisibleLayers = useMemo(() => {
  return visibleLayers.map((layer) => ({
    ...layer,
    data: {
      ...layer.data,
      features: layer.data.features.filter((feature) => {
        const level2Id = String(feature.properties.level2Id ?? "").trim();
        const level3Id = String(feature.properties.level3Id ?? "").trim();
        const level4Id = String(feature.properties.level4Id ?? "").trim();

        return (
          (!selectedLevel2 || level2Id === selectedLevel2) &&
          (!selectedLevel3 || level3Id === selectedLevel3) &&
          (!selectedLevel4 || level4Id === selectedLevel4)
        );
      })
    }
  }));
}, [visibleLayers, selectedLevel2, selectedLevel3, selectedLevel4]);
```

### Text Search

Search by code or name:

```typescript src/components/GeoDashboardMapClient.tsx
const searchMatches = useMemo(() => {
  if (!normalizedSearch) return [];

  const matches = [];
  for (const layer of layers) {
    for (const feature of layer.data.features) {
      const level4Code = String(feature.properties.level4Code ?? "");
      const level4Name = String(feature.properties.level4Name ?? "");
      
      if (
        level4Code.toLowerCase().includes(normalizedSearch) ||
        level4Name.toLowerCase().includes(normalizedSearch)
      ) {
        matches.push({
          level4Id: feature.properties.level4Id,
          label: `${level4Code} · ${level4Name}`,
          geometry: feature.geometry
        });
      }
    }
  }
  return matches;
}, [layers, normalizedSearch]);
```

## Performance Optimization

<AccordionGroup>
  <Accordion title="Client-Side Rendering">
    The map component is dynamically imported with `ssr: false` to prevent server-side rendering issues with Leaflet.
  </Accordion>
  
  <Accordion title="Layer Caching">
    Layer settings (visibility, opacity, color) are persisted to localStorage to maintain state across sessions.
  </Accordion>
  
  <Accordion title="Memoized Computations">
    Heavy computations like filtering, area calculations, and search results are memoized with `useMemo`.
  </Accordion>
  
  <Accordion title="Lazy Feature Loading">
    Features are loaded with bounding box filtering to limit the amount of data transferred.
  </Accordion>
</AccordionGroup>

<Note>
  The geospatial system uses EPSG:4326 (WGS 84) coordinate reference system for all spatial data.
</Note>
